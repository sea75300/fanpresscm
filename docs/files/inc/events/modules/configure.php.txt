class configure extends \fpcm\events\abstracts\eventReturnArray {

    /**
     * 
     * @return array
     */
    public function run()
    {
        $class = \fpcm\module\module::getEventNamespace($this->data, $this->getEventClassBase());
        if (!class_exists($class)) {
            return true;
        }

        $obj = new $class($this->data);
        if (!$this->is_a($obj)) {
            return false;
        }

        return $obj->run();
    }

}
